select tb_mahasiswa.mhs_nama, tb_matakuliah.mk_nama, tb_matakuliah.mk_code, max(tb_mahasiswa_nilai.nilai) as nilai from tb_mahasiswa_nilai join tb_matakuliah on tb_mahasiswa_nilai.mk_id = tb_matakuliah.mkid join tb_mahasiswa on tb_mahasiswa_nilai.mhs_id = tb_mahasiswa.mhs_id where tb_matakuliah.mk_kode = "MK303";
